# 1.选redis还是gateway

做接口限流，这里偏向于业务限流，具体的业务，所以用redis，gateway的粒度大，是对所有接口进行业务网关控制，这里还是上redis限流更好

Gateway 限流：偏通用流量保护
业务层限流：偏业务规则保护

Nginx：挡最外层流量
Gateway：做微服务网关规则
业务服务：做具体业务限流

# 2.RedisScriptConfig

```java
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import org.springframework.core.io.ClassPathResource;
import org.springframework.data.redis.core.script.DefaultRedisScript;

/**
 * lua脚本加载配置类
 */
@Configuration
public class RedisScriptConfig {

    @Bean("fixedWindowRateLimitScript")
    public DefaultRedisScript<Long> fixedWindowRateLimitScript() {
        DefaultRedisScript<Long> script = new DefaultRedisScript<>();
        script.setLocation(new ClassPathResource("lua/fixed-window-rate-limit.lua"));
        script.setResultType(Long.class);
        return script;
    }
}
```

# 3.RedisFixedWindowRateLimiter

```java
import com.haoclass.common.exception.BusinessException;
import lombok.RequiredArgsConstructor;
import org.springframework.data.redis.core.StringRedisTemplate;
import org.springframework.data.redis.core.script.DefaultRedisScript;
import org.springframework.stereotype.Component;

import java.time.Duration;
import java.util.ArrayList;
import java.util.List;

/**
 * 加载redis限流脚本
 */
@Component
@RequiredArgsConstructor
public class RedisFixedWindowRateLimiter {

    private static final String KEY_PREFIX = "haoclass:rate-limit:";

    private final StringRedisTemplate stringRedisTemplate;

    private final DefaultRedisScript<Long> fixedWindowRateLimitScript;

    /**
     *
     * @param key 限流对象key
     * @param maxCount 允许最大次数
     * @param window 限流窗口时间
     */
    public void check(String key, int maxCount, Duration window) {
        List<String> keys = new ArrayList<>();
        keys.add(KEY_PREFIX + key);
        Long result = stringRedisTemplate.execute(fixedWindowRateLimitScript,
                keys,
                String.valueOf(maxCount),
                String.valueOf(window.toSeconds()));

        if (result == null || result == 0) {
            throw BusinessException.badRequest(
                    "操作过于频繁，请稍后再试"
            );
        }
    }
}
```

# 3.Lua脚本

```lua
local key = KEYS[1]
local maxCount = tonumber(ARGV[1])
local windowSeconds = tonumber(ARGV[2])

local current = redis.call('INCR', key)

if current == 1 then
    redis.call('EXPIRE', key, windowSeconds)
end

if current > maxCount then
    return 0
end

return 1
```
