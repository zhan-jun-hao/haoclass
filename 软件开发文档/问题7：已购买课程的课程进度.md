# 1.怎么选择连接方式

LEFT JOIN：当主表关联的表里没有对应的数据，那就会返回NULL

INNER JOIN：当主表关联的表里没有对应数据，就不会返回，连NULL也没有，都不返回

# 2.怎么写出SQL

想拼接数据，不用想表结构

```sql
<?xml version="1.0" encoding="UTF-8" ?>
<!DOCTYPE mapper
        PUBLIC "-//mybatis.org//DTD Mapper 3.0//EN"
        "http://mybatis.org/dtd/mybatis-3-mapper.dtd">

<mapper namespace="com.haoclass.main.infrastructure.persistence.mapper.ClientMyCourseMapper">

    <resultMap id="ClientMyCourseVoMap" type="com.haoclass.main.interfaces.response.client.ClientMyCourseVo">
        <result column="courseId" property="courseId"/>
        <result column="title" property="title"/>
        <result column="coverUrl" property="coverUrl"/>
        <result column="teacherName" property="teacherName"/>
        <result column="episodeCount" property="episodeCount"/>
        <result column="finishedEpisodeCount" property="finishedEpisodeCount"/>
        <result column="learningPercent" property="learningPercent"/>
        <result column="lastEpisodeId" property="lastEpisodeId"/>
        <result column="lastEpisodeTitle" property="lastEpisodeTitle"/>
        <result column="lastProgressSeconds" property="lastProgressSeconds"/>
        <result column="lastLearnTime" property="lastLearnTime"/>
        <result column="grantTime" property="grantTime"/>
        <result column="expireTime" property="expireTime"/>
    </resultMap>

    <select id="selectMyCourses" resultMap="ClientMyCourseVoMap">
        SELECT
            c.id AS courseId,
            c.title,
            c.coverUrl,
            c.teacherName,
            c.episodeCount,
            IFNULL(fp.finishedEpisodeCount, 0) AS finishedEpisodeCount,
            CASE
                WHEN IFNULL(c.episodeCount, 0) = 0 THEN 0
                ELSE FLOOR(IFNULL(fp.finishedEpisodeCount, 0) * 100 / c.episodeCount)
            END AS learningPercent,
            lp.episodeId AS lastEpisodeId,
            ce.title AS lastEpisodeTitle,
            IFNULL(lp.progressSeconds, 0) AS lastProgressSeconds,
            lp.lastLearnTime,
            cu.grantTime,
            cu.expireTime
        FROM course_user cu

        INNER JOIN course c
            ON c.id = cu.courseId
            AND c.deleted = 0

        LEFT JOIN (
            SELECT
                courseId,
                COUNT(*) AS finishedEpisodeCount
            FROM course_learning_progress
            WHERE userId = #{userId}
                AND finished = 1
                AND deleted = 0
            GROUP BY courseId
        ) fp
            ON fp.courseId = c.id

        LEFT JOIN course_learning_progress lp
        ON lp.id = (
        SELECT p.id
            FROM course_learning_progress p
            WHERE p.userId = #{userId}
                AND p.courseId = c.id
                AND p.deleted = 0
            ORDER BY p.lastLearnTime DESC, p.updateTime DESC, p.id DESC
            LIMIT 1
        )

        LEFT JOIN course_episode ce
            ON ce.id = lp.episodeId
            AND ce.courseId = c.id
            AND ce.deleted = 0
        WHERE cu.userId = #{userId}
            AND cu.deleted = 0
            AND cu.status = 1
            AND (cu.expireTime IS NULL OR cu.expireTime &gt; NOW())
        ORDER BY COALESCE(lp.lastLearnTime, cu.grantTime) DESC
    </select>
</mapper>

```
