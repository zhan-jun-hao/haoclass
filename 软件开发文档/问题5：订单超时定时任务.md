对于@Schedule和xxljob的选择

```java
少量简单任务             @Scheduled + RLock
任务逐渐增多             XXL-JOB
需要管理后台、日志、重试  XXL-JOB
需要手动触发、暂停任务     XXL-JOB
需要故障转移和告警         XXL-JOB
```

```java
import com.haoclass.main.application.service.inner.InnerCourseOrderService;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.redisson.api.RLock;
import org.redisson.api.RedissonClient;
import org.springframework.scheduling.annotation.Scheduled;
import org.springframework.stereotype.Component;
import java.time.LocalDateTime;
import java.util.concurrent.TimeUnit;

@Slf4j
@Component
@RequiredArgsConstructor
public class CloseExpiredPendingOrders {

    private final InnerCourseOrderService innerCourseOrderApplicationService;
    private final RedissonClient redissonClient;

    @Scheduled(cron = "0 * * * * ?")
    public void closeExpiredPendingOrders() {
        String lockKey = "main-service:order:schedule:close-expired-orders";
        RLock lock = redissonClient.getLock(lockKey);
        try {
            boolean isLocked = lock.tryLock(0, 60,TimeUnit.SECONDS);
            if(!isLocked) {
                return;
            }

            int columns = innerCourseOrderApplicationService.closeExpiredPendingOrders(LocalDateTime.now());
            log.info("关闭订单定时任务:{}条", columns);

        } catch (InterruptedException e) {
            // 标记该线程已经中断
            Thread.currentThread().interrupt();
            log.warn("关闭过期订单定时任务已中断", e);
        } finally {
            if(lock.isHeldByCurrentThread()) {
                lock.unlock();
            }
        }
    }
}
```
